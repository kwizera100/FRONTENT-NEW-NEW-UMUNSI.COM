// This is your Prisma schema file,
// learn more about it in the docs: https://pris.ly/d/prisma-schema

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model User {
  id         String   @id @default(cuid())
  email      String   @unique
  username   String   @unique
  password   String
  firstName  String?
  lastName   String?
  role       String @default("USER")
  isPremium  Boolean  @default(false)
  premiumSince DateTime?
  premiumUntil DateTime?
  avatar     String?
  bio        String?
  phone      String?
  profileUrl String?
  isActive   Boolean  @default(true)
  isVerified Boolean  @default(false)
  lastLogin  DateTime?
  createdAt  DateTime @default(now())
  updatedAt  DateTime @updatedAt

  // Relations
  news News[]
  mediaFiles MediaFile[]
  posts Post[]
  supportPayments SupportPayment[]
  premiumPostAccess PremiumPostAccess[]
  classifiedAds ClassifiedAd[] @relation("ClassifiedAuthor")
  reviewedClassifiedAds ClassifiedAd[] @relation("ClassifiedReviewer")
  classifiedBroadcasts ClassifiedBroadcast[] @relation("ClassifiedBroadcastCreator")

  @@map("users")
}

model ClassifiedAd {
  id             String   @id @default(cuid())
  userId         String
  reviewedById   String?
  category       String
  title          String
  description    String
  phone          String
  email          String
  attachmentName String?
  attachmentUrl  String?
  durationDays   Int
  priceRwf       Int
  status         String   @default("PENDING")
  reviewNote     String?
  createdAt      DateTime @default(now())
  updatedAt      DateTime @updatedAt

  user       User  @relation("ClassifiedAuthor", fields: [userId], references: [id], onDelete: Cascade)
  reviewedBy User? @relation("ClassifiedReviewer", fields: [reviewedById], references: [id], onDelete: SetNull)

  @@index([userId])
  @@index([category])
  @@index([status])
  @@map("classified_ads")
}

model ClassifiedBroadcast {
  id          String   @id @default(cuid())
  message     String
  createdById String
  createdAt   DateTime @default(now())

  createdBy User @relation("ClassifiedBroadcastCreator", fields: [createdById], references: [id], onDelete: Cascade)

  @@index([createdAt])
  @@map("classified_broadcasts")
}

model SupportPayment {
  id               String   @id @default(cuid())
  userId           String
  provider         String   @default("KPAY")
  purpose          String   @default("PREMIUM_SUPPORT")
  amount           Float
  currency         String   @default("RWF")
  status           String   @default("PENDING")
  txRef            String   @unique
  flwTransactionId String?
  checkoutUrl      String?
  providerStatus   String?
  customerEmail    String?
  paidAt           DateTime?
  errorMessage     String?
  metadata         String?
  createdAt        DateTime @default(now())
  updatedAt        DateTime @updatedAt

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@map("support_payments")
}

model Category {
  id          String   @id @default(cuid())
  name        String   @unique
  slug        String   @unique
  description String?
  color       String?
  icon        String?
  isActive    Boolean  @default(true)
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  // Relations
  news News[]
  posts Post[]

  @@map("categories")
}

model News {
  id            String      @id @default(cuid())
  title         String
  slug          String      @unique
  content       String
  excerpt       String?
  featuredImage String?
  status        String  @default("DRAFT")
  publishedAt   DateTime?
  viewCount     Int         @default(0)
  likeCount     Int         @default(0)
  isFeatured    Boolean     @default(false)
  isBreaking    Boolean     @default(false)
  isTrending    Boolean     @default(false)
  createdAt     DateTime    @default(now())
  updatedAt     DateTime    @updatedAt

  // Relations
  authorId    String
  author      User      @relation(fields: [authorId], references: [id], onDelete: Cascade)
  categoryId  String?
  category    Category? @relation(fields: [categoryId], references: [id])

  @@map("news")
}

model MediaFile {
  id            String   @id @default(cuid())
  filename      String
  originalName  String
  mimeType      String
  size          Int
  url           String
  thumbnailUrl  String?
  category      String
  tags          String
  description   String?
  isPublic      Boolean  @default(true)
  isFeatured    Boolean  @default(false)
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt

  // Relations
  uploadedById  String
  uploadedBy    User     @relation(fields: [uploadedById], references: [id], onDelete: Cascade)

  @@map("media_files")
}

model Post {
  id            String      @id @default(cuid())
  title         String
  slug          String      @unique
  content       String
  excerpt       String?
  featuredImage String?
  status        String  @default("DRAFT")
  isPremium     Boolean     @default(false)
  publishedAt   DateTime?
  viewCount     Int         @default(0)
  likeCount     Int         @default(0)
  commentCount  Int         @default(0)
  isFeatured    Boolean     @default(false)
  isPinned      Boolean     @default(false)
  allowComments Boolean     @default(true)
  tags          String
  metaTitle     String?
  metaDescription String?
  createdAt     DateTime    @default(now())
  updatedAt     DateTime    @updatedAt

  // Relations
  authorId    String
  author      User      @relation(fields: [authorId], references: [id], onDelete: Cascade)
  categoryId  String?
  category    Category? @relation(fields: [categoryId], references: [id])
  premiumPostAccess PremiumPostAccess[]

  @@map("posts")
}

model PremiumPostAccess {
  id         String   @id @default(cuid())
  userId     String
  postId     String
  grantedBy  String?
  expiresAt  DateTime?
  createdAt  DateTime @default(now())
  updatedAt  DateTime @updatedAt

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)
  post Post @relation(fields: [postId], references: [id], onDelete: Cascade)

  @@unique([userId, postId])
  @@index([userId])
  @@index([postId])
  @@map("premium_post_access")
}
